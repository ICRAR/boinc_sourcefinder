# BOINC VM Initialization
echo BOINC VM Initialization [Sleeping 5 seconds]
sleep 5

echo -- Mounting shared directory
mount -t vboxsf shared /root/shared
if [ $? -ne 0 ]; then
    echo ---- Failed to mount shared directory
    sleep 5
    shutdown -hP 0
fi

echo -- Launching Application
if [ -f /root/boinc_app ]; then
    mkdir -p /root/worker
    cp boinc_app /root/worker/
    cp -r /root/shared/* /root/worker/
    cd root/worker/
    echo --- Application has been moved the the worker directory
    sleep 5
    ./boinc_app |& vboxmonitor
    echo --- Application and monitor have started
    cd /root
    cp -r /root/worker/* /root/shared/
    cp /root/shared/boinc_app /root/
    rm -rf /root/worker
    echo --- Everything has gone according to plan
    sleep 5
    shutdown -hP 0
else
    echo ---- Failed to launch script
    sleep 5
fi
shutdown -hP 0
